1. 💨 Project Introduction

Gas Detector project mein Arduino Uno ek MQ-2 gas sensor se LPG, smoke, methane, aur hydrogen detect karta hai. Jab gas level ek threshold cross kare toh buzzer bajta hai aur LED jal jaati hai — bilkul ek real gas leak alarm ki tarah! Sensor ka analog value 0–1023 range mein aata hai.

💡 Important: MQ-2 sensor ko use karne se pehle 2–3 minute warm-up dena zaroori hai — tabhi accurate readings milti hain. Cold start pe readings galat ho sakti hain!

2. 📋 Components Required

  • 🟩 Arduino Uno × 1

  • 💡 Red LED × 1 (gas alert indicator)

4. 💻 Arduino Code

⚠️ Library: Koi extra library nahi chahiye — standard analogRead() aur digitalRead() use hoga! 
// ============================================
// Gas Detector Project - Arduino Uno
// MQ-2 Sensor | Buzzer | LED Alert
// AO → A0 | DO → Pin 7 | Buzzer → Pin 8 | LED → Pin 9
// ============================================

const int mq2Analog  = A0;  // MQ-2 Analog output
const int mq2Digital = 7;   // MQ-2 Digital output
const int buzzerPin  = 8;   // Buzzer pin
const int ledPin     = 9;   // Alert LED pin

int  gasLevel   = 0;        // Analog gas value (0-1023)
int  threshold  = 400;      // Gas threshold — yahan se alarm bajega
bool alarmOn    = false;

// ============================================
void setup() {
  pinMode(mq2Digital, INPUT);
  pinMode(buzzerPin,  OUTPUT);
  pinMode(ledPin,     OUTPUT);
  Serial.begin(9600);
  Serial.println("=== Gas Detector Ready ===");
  Serial.println("Sensor warm-up... (2 min wait karo)");
  delay(2000);  // Sensor warm-up time
  Serial.println("Monitoring started!");
}

// ============================================
void loop() {
  // Analog gas level padho (0 = clean air, 1023 = very high gas)
  gasLevel = analogRead(mq2Analog);

  // Digital threshold (MQ-2 ka onboard comparator)
  int digitalVal = digitalRead(mq2Digital);

  Serial.print("Gas Level: ");
  Serial.print(gasLevel);
  Serial.print(" / 1023  |  Status: ");

  if (gasLevel > threshold || digitalVal == HIGH) {
    // ---- GAS DETECTED — ALARM ON! ----
    alarmOn = true;
    Serial.println("🚨 GAS DETECTED! ALARM ON!");

    digitalWrite(ledPin, HIGH);       // LED ON
    digitalWrite(buzzerPin, HIGH);    // Buzzer ON
    delay(500);
    digitalWrite(buzzerPin, LOW);     // Buzzer beep pattern
    delay(200);
    digitalWrite(buzzerPin, HIGH);
    delay(500);
    digitalWrite(buzzerPin, LOW);

  } else {
    // ---- Safe — No Gas ----
    alarmOn = false;
    Serial.println("✅ Safe — No Gas");

    digitalWrite(ledPin,    LOW);     // LED OFF
    digitalWrite(buzzerPin, LOW);     // Buzzer OFF
  }

  // Gas level percentage print karo
  int percent = map(gasLevel, 0, 1023, 0, 100);
  Serial.print("Gas %: ");
  Serial.print(percent);
  Serial.println("%");
  Serial.println("-------------------");

  delay(1000);  // Har 1 second mein check karo
}

// ============================================
// CALIBRATION TIPS:
// 1. Sensor ko fresh open air mein rakho
// 2. Serial Monitor mein baseline value note karo (~100-200)
// 3. threshold = baseline + 150 set karo best accuracy ke liye
// 4. LPG/Smoke test: lighter (bina flame) sensor ke paas lo

7. 🛠️ Common Errors & Fix

  • Value hamesha high? → Sensor warm-up nahi hua — 2-3 min wait karo open air mein
  • Buzzer nahi baja? → Pin 8 check karo, active buzzer use karo (passive nahi)
  • LED nahi jali? → LED polarity check karo (anode/cathode) + 220Ω resistor lagao
  • Value 0 aa rahi? → A0 pin connection aur VCC/GND check karo
  • False alarm baar baar? → threshold value badhaao code mein (400 → 500 ya 600)
  • Reading fluctuate karti? → 3 readings average karo; capacitor 100nF A0 aur GND ke beech lagao
