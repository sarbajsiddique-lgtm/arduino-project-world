1. 🌱 Project Introduction

Soil Moisture project mein Arduino Uno ek capacitive/resistive soil moisture sensor se mitti ki naami (moisture) measure karta hai. Agar mitti dry ho jaaye toh buzzer beep karta hai ya relay se water pump ON ho jaata hai — bilkul automatic watering system! Sensor ka analog value 0–1023 range mein aata hai jise percentage mein convert karte hain.

💡 Fun Fact: Capacitive soil moisture sensor resistive se zyada accurate hota hai aur kharaab bhi nahi hota — kyunki uske metal prongs soil ke directly contact mein nahi aate!

4. 💻 Arduino Code

📦 Library: Koi extra library zarurat nahi — standard analogRead() use hoga! 
// ============================================
// Soil Moisture Monitor - Arduino Uno
// Sensor AOUT → A0 | Buzzer → Pin 8
// Green LED → Pin 9 | Red LED → Pin 10
// ============================================

#define MOISTURE_PIN  A0   // Analog sensor pin
#define DOUT_PIN      2    // Digital out (optional)
#define BUZZER_PIN    8    // Buzzer alert
#define GREEN_LED     9    // Wet = green ON
#define RED_LED       10   // Dry = red ON

// ---- Calibration values ----
// In air (completely dry): ~1023
// In water (completely wet): ~300
// Adjust these after testing your sensor!
#define DRY_VALUE   850    // Isse upar = dry
#define WET_VALUE   400    // Isse neeche = wet
#define WARN_VALUE  650    // Warning zone

int  rawValue    = 0;
int  moisturePct = 0;

// ============================================
void setup() {
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(GREEN_LED,  OUTPUT);
  pinMode(RED_LED,    OUTPUT);
  pinMode(DOUT_PIN,   INPUT);

  Serial.begin(9600);
  Serial.println("=== Soil Moisture Monitor Ready ===");
  Serial.println("Sensor mitti mein daalo aur readings dekho!");
  delay(1000);
}

// ============================================
void loop() {
  // Analog value padhna (0 = wet, 1023 = dry)
  rawValue = analogRead(MOISTURE_PIN);

  // Percentage mein convert (0% dry, 100% wet)
  moisturePct = map(rawValue, DRY_VALUE, WET_VALUE, 0, 100);
  moisturePct = constrain(moisturePct, 0, 100);

  // Serial Monitor output
  Serial.println("================================");
  Serial.print("💧 Raw Value    : ");
  Serial.println(rawValue);
  Serial.print("💧 Moisture     : ");
  Serial.print(moisturePct);
  Serial.println(" %");

  // ---- Status check ----
  if (moisturePct >= 60) {
    // WET - Plant khush hai!
    digitalWrite(GREEN_LED, HIGH);
    digitalWrite(RED_LED,   LOW);
    digitalWrite(BUZZER_PIN, LOW);
    Serial.println("✅ Status: WET — Plant is happy!");

  } else if (moisturePct >= 30) {
    // MEDIUM - Theek hai
    digitalWrite(GREEN_LED, LOW);
    digitalWrite(RED_LED,   LOW);
    digitalWrite(BUZZER_PIN, LOW);
    Serial.println("⚠️ Status: MEDIUM — Monitor karo");

  } else {
    // DRY - Paani chahiye!
    digitalWrite(GREEN_LED, LOW);
    digitalWrite(RED_LED,   HIGH);
    // Buzzer beep karo
    tone(BUZZER_PIN, 1000, 200);
    Serial.println("🚨 Status: DRY — Paani do plant ko!");
  }

  Serial.println("================================");
  delay(2000); // Har 2 second mein read karo
}

// ============================================
// BONUS: Moisture bar Serial mein print karo
void printMoistureBar(int pct) {
  Serial.print("[");
  int filled = pct / 5; // 20 blocks total
  for(int i = 0; i < 20; i++){
    Serial.print(i < filled ? "█" : "░");
  }
  Serial.print("] ");
  Serial.print(pct);
  Serial.println("%");
}

7. 🛠️ Common Errors & Fix

  • Always 0% ya 100%? → DRY_VALUE aur WET_VALUE calibrate karo apne sensor se
  • Reading fluctuate karti hai? → Solder connections check karo; 3 readings average karo
  • Sensor kharaab ho gaya? → Resistive (YL-69) jaldi kharaab hote hain — capacitive use karo
  • Buzzer nahi baj raha? → Active buzzer use karo ya tone() function check karo
  • LED nahi jal rahi? → LED polarity (anode/cathode) aur 220Ω resistor check karo
